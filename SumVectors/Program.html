package application; import java.util.Locale; import java.util.Scanner; public class Program { public static void main(String[] args) { Locale.setDefault(Locale.US); Scanner sc = new Scanner(System.in); System.out.print("How many numbers you will be entered? "); int n = sc.nextInt(); double[] vect = new double[n]; for(int i = 0; i < vect.length; i ++) { System.out.print("Enter a number: "); double num = sc.nextDouble(); vect[i] = num; } double sum = 0.0, cont = 0.0; for(int i = 0; i < vect.length; i++) { sum += vect[i]; cont++; } Double average = sum / cont; System.out.println(); System.out.print("Values = "); for(int i = 0; i < vect.length; i++) { System.out.print(vect[i] + " "); } System.out.println(); System.out.printf("Sum = %.2f%n", sum); System.out.printf("Average = %.2f", average); sc.close(); } }
