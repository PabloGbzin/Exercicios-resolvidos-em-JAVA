package application; import java.util.Scanner; import java.util.Locale; import entities.Calc; public class Program { public static void main(String[] args) { Locale.setDefault(Locale.US); Scanner sc = new Scanner(System.in); System.out.print("Enter radius: "); double radius = sc.nextDouble(); System.out.printf("Circumference: %.2f%n", Calc.circum(radius)); System.out.printf("Volume: %.2f%n", Calc.vol(radius)); System.out.printf("PI value: %.2f", Calc.pi); sc.close(); } }
