package application; import java.util.Locale; import java.util.Scanner; import entities.BelowAverage; public class Program { public static void main(String[] args) { Locale.setDefault(Locale.US); Scanner sc = new Scanner(System.in); System.out.print("How many elements will be the vector? "); int n = sc.nextInt(); BelowAverage[] vect = new BelowAverage[n]; for(int i = 0; i < vect.length; i ++) { System.out.print("Enter a number: "); Double num = sc.nextDouble(); vect[i] = new BelowAverage(num); } Double sum = 0.0, average = 0.0; for(int i = 0; i < vect.length; i++) { sum += vect[i].getNotas(); } average = sum / vect.length; System.out.println(); System.out.printf("Vector average = %.3f%n", average); System.out.println("Elements average below: "); for(int i = 0; i < vect.length; i ++) { if(vect[i].getNotas() < average) { System.out.println(vect[i].getNotas()); } } sc.close(); } }
